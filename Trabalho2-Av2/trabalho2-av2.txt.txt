2-Reescreva o codigo abaixo usando a estrutura for.
#!/bin/bash
cont=$1
fat=1
while [ $cont -gt 0 ] 
do
  fat=$(( $fat * $cont ))
  cont=$(( $cont - 1 ))
done
echo $fat

RESPOSTA: Código reescrito com FOR

#!/bin/bash

cont=$1

fat=1


for ((i=1; i <= $cont ; i++))

do
    
fat=$(($fat*$i))

done
